1. 設定`BioStarProvisioningQueue`
```C#
using System.Threading.Channels;

namespace HaikouAPI.Service.BioStar
{
	/// <summary>
    /// 設備端人員新增/複製/刪除的全域單一佇列（singleton）。
    /// API 端驗證通過後 Enqueue，BioStarProvisioningBackgroundService 依序取出處理，
    /// 避免對設備端下指令的緩慢呼叫卡住 HTTP 請求，也避免同一設備的 UserId 計算因併發而衝突。
    /// </summary>
    public sealed class BioStarProvisioningQueue
    {
        private readonly Channel<BioStarProvisioningJob> _channel =
            Channel.CreateUnbounded<BioStarProvisioningJob>(new UnboundedChannelOptions { SingleReader = true });

        public void Enqueue(BioStarProvisioningJob job) => _channel.Writer.TryWrite(job);

        public ChannelReader<BioStarProvisioningJob> Reader => _channel.Reader;
    }

    /// <summary>
    /// 在設備端新增/複製/刪除人員的背景工作項目。
    /// </summary>
    public abstract record BioStarProvisioningJob;

	//以下是宣告InsertUserJob、CopyUserJob、DeleteUserJob
    public sealed record InsertUserJob(Guid PlaceholderId, long PermissionId, UInt32 TargetDeviceId) : BioStarProvisioningJob;

    public sealed record CopyUserJob(Guid PlaceholderId, Guid SourceUserRelatedId, UInt32 TargetDeviceId) : BioStarProvisioningJob;

    public sealed record DeleteUserJob(Guid UserRelatedId) : BioStarProvisioningJob; 
}
```

2. 在`Program.cs` 將 `BioStarProvisioningQueue`註冊為`Singleton`：
```C#
builder.Services.AddSingleton<BioStarProvisioningQueue>();
```

3. 設定`BioStarProvisioningBackgroundService`
```C#
using HaikouAPI.Hubs;
using Microsoft.AspNetCore.SignalR;

namespace HaikouAPI.Service.BioStar.Background
{
    /// <summary>
    /// 依序處理 BioStarProvisioningQueue 中「在設備端新增/複製/刪除人員」的工作，
    /// 把原本會卡住 HTTP 請求的設備端 SDK 呼叫移到背景全域單一佇列依序執行。
    /// 每筆工作處理完後，透過 BioStarProvisioningHub 推播成功/失敗結果給前端。
    /// </summary>
    public class BioStarProvisioningBackgroundService : BackgroundService
    {
        private readonly IServiceProvider _serviceProvider;
        private readonly BioStarProvisioningQueue _queue;
        private readonly ILogger<BioStarProvisioningBackgroundService> _logger;

        public BioStarProvisioningBackgroundService(
        IServiceProvider serviceProvider, 
        BioStarProvisioningQueue queue,
        ILogger<BioStarProvisioningBackgroundService> logger)
        {
            _serviceProvider = serviceProvider;
            _queue = queue;
            _logger = logger;
        }

        protected override async Task ExecuteAsync(CancellationToken stoppingToken)
        {
            await foreach (var job in _queue.Reader.ReadAllAsync(stoppingToken))
            {
                var jobTypeName = job.GetType().Name;
                var relatedId = GetRelatedId(job);
                bool success = true;
                string? errorMessage = null;

                try
                {
                    using var scope = _serviceProvider.CreateScope();
                    var service = scope.ServiceProvider.GetRequiredService<DoorAccessByBioStarService>();

                    switch (job)
                    {
                        case InsertUserJob insertJob:
                            await service.ProcessInsertUserJobAsync(insertJob, stoppingToken);
                            break;
                        case CopyUserJob copyJob:
                            await service.ProcessCopyUserJobAsync(copyJob, stoppingToken);
                            break;
                        case DeleteUserJob deleteJob:
                            await service.ProcessDeleteUserJobAsync(deleteJob, stoppingToken);
                            break;
                    }
                }
                catch (Exception ex) when (ex is not OperationCanceledException)
                {
                    success = false;
                    errorMessage = ex.Message;
                    _logger.LogError(ex, "背景執行設備端人員新增/複製/刪除工作時發生錯誤：{JobType}", jobTypeName);
                }
            }
        }

        private static Guid GetRelatedId(BioStarProvisioningJob job) => job switch
        {
            InsertUserJob j => j.PlaceholderId,
            CopyUserJob j => j.PlaceholderId,
            DeleteUserJob j => j.UserRelatedId,
            _ => Guid.Empty
        };
    }
}
```

4. 主程式呼叫排入背景佇列
```C#
public class DoorAccessByBioStarService
{
	private readonly BioStarProvisioningQueue _provisioningQueue;
	
	public DoorAccessByBioStarService(BioStarProvisioningQueue provisioningQueue)
	{
	    _provisioningQueue = provisioningQueue;
	}
	
	public async Task InsertUserToNewDeviceAsync(long PermissionId, UInt32 targetDeviceId)
	{
		//帶入參數值
		_provisioningQueue.Enqueue(new InsertUserJob(placeholder.Id, PermissionId, targetDeviceId));
	}
	
	public async Task CopyBioStarUserRelatedsToNewDeviceAsync(Guid id, UInt32 targetDeviceId)
	{
		//帶入參數值
		_provisioningQueue.Enqueue(new CopyUserJob(placeholder.Id, id, targetDeviceId));
	}
	
	public async Task DeleteUserToDeviceAsync(Guid id)
	{
		//帶入參數值
		_provisioningQueue.Enqueue(new DeleteUserJob(id));
	}
}
```
